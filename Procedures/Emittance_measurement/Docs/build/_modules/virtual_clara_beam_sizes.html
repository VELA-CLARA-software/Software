        self.catapdict = self.machinestate.getCATAPDict(self.mode)
        get_data_from_catap = self.machinestate.getDataFromCATAP()
        for name in self.catapdict['Magnet'].keys():

        self.catapdata = self.machinestate.getMachineStateFromCATAP(self.mode)

        self.machinestate.exportParameterValuesToYAMLFile("catap-test.yaml", self.catapdata)


[docs]    def preparing_simframe(self):
        self.machinestate.getMachineStateFromSimFrame('test', 'Lattices/CLA10-BA1_OM.def')
        self.machinestate.initialiseSimFrameData()
        self.simframedata = self.machinestate.getSimFrameDataDict()
        self.framework = self.machinestate.getFramework()
        # Read in machine state .yml file. (Obviously you can do this directly from line 35 above)
        machinestatefile = self.machinestate.parseParameterInputFile("catap-test.yaml")

        # Set up lattice end point (default is to start at generator, at the moment we have to do it this way)
        self.machinestate.updateLatticeEnd(machinestatefile, 'CLA-S02')

        self.catapdata = self.machinestate.getMachineStateFromCATAP(self.mode)
        # We have to fudge this for now as the conversion between current and solenoid strength doesn't work
        self.catapdata['INJ']['CLA-LRG1-MAG-SOL-01'].update({'field_amplitude': 0.255})

        self.catapdata['INJ']['CLA-LRG1-MAG-BSOL-01'].update({'field_amplitude': -0.07})
